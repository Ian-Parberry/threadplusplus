46const std::string to_string(const float x, const size_t n){
47 std::ostringstream out;
48 out.precision(n);
49 out << std::fixed << x;
50
51 std::string s = out.str(); //s contains x with precision n
52
53 //find last non-zero, non-decimal-point digit of s
54
55 int i = (int)s.length() - 1; //index
56
57 while(i >= 0 && (s[i] == '0' || s[i] == '.'))
58 i--;
59
60 //truncate x to last non-zero, non-decimal-point digit
61
62 return s.substr(0, i + 1);
63} //to_string

92const std::string CTimer::TimeString(float fSeconds, size_t n) const{
93 std::string s; //for the time string, initially empty
94
95 constexpr size_t SECONDS_PER_MINUTE = 60;
96 constexpr size_t SECONDS_PER_HOUR = 60*SECONDS_PER_MINUTE;
97 constexpr size_t SECONDS_PER_DAY = 24*SECONDS_PER_HOUR;
98
99 //number of days
100
101 const size_t nDays = (size_t)floor(fSeconds/SECONDS_PER_DAY);
102 fSeconds -= nDays*SECONDS_PER_DAY;
103
104 if(nDays > 0)
105 s += std::to_string(nDays) + (nDays > 1? " days": " day");
106
107 //number of hours
108
109 const size_t nHours = (size_t)floor(fSeconds/SECONDS_PER_HOUR);
110 fSeconds -= nHours*SECONDS_PER_HOUR;
111
112 if(nHours > 0){
113 AppendCommaSeparator(s);
114 s += std::to_string(nHours) + " hr";
115 } //if
116
117 //number of minutes
118
119 const size_t nMinutes = (size_t)floor(fSeconds/SECONDS_PER_MINUTE);
120 fSeconds -= nMinutes*SECONDS_PER_MINUTE;
121
122 if(nMinutes > 0){
123 AppendCommaSeparator(s);
124 s += std::to_string(nMinutes) + " min";
125 } //if
126
127 AppendCommaSeparator(s);
128
129 //number of seconds
130
131 if(nMinutes >= 1){
132 const size_t nSeconds = size_t(std::ceil(fSeconds));
133 if(nSeconds > 0)
134 s += std::to_string(nSeconds) + " sec";
135 } //if
136
137 else if(fSeconds >= 1.0f)
138 s += to_string(fSeconds, n) + " sec";
139
140 //number of milliseconds when less than 1 second
141
142 else if(fSeconds >= 0.001f)
143 s += std::to_string(size_t(std::ceilf(fSeconds*1000.0f))) + " ms";
144
145 else s += "< 1 ms"; //less than a millisecond, so whatever
146
147 //s now contains a properly formatted time string
148
149 return s;
150} //TimeString

192const uint64_t CTimer::CPUTime() const{
193#if defined(_MSC_VER) //Windows Visual Studio
194 uint64_t llCNS = 0; //for CPU time in centinanoseconds
195
196 HANDLE hProcess = //handle to this process
197 OpenProcess(PROCESS_QUERY_INFORMATION, FALSE, GetCurrentProcessId());
198
199 if(hProcess != nullptr){ //handle is valid
200 FILETIME ftCreation, ftExit, ftKernel; //unused
201 FILETIME ftUser; //this one we want
202
203 if(GetProcessTimes(hProcess, &ftCreation, &ftExit, &ftKernel, &ftUser))
204 llCNS = *((uint64_t*)&ftUser); //get time values
205
206 CloseHandle(hProcess); //close the process handle
207 } //if
208
209 return uint64_t(llCNS)/10000;
210#else //g++, *nix
211 return uint64_t(1000.0*(double)(clock())/CLOCKS_PER_SEC);
212#endif
213} //CPUTime
