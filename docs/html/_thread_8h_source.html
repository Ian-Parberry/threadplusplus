68template <class CTaskClass>
69void CThread<CTaskClass>::operator()(){
70 bool bActive = true; //true to stay active, false to exit thread
71
72 while(bActive){ //perform task loop
73 CTaskClass* pTask = nullptr; //current task descriptor
74
75 if(CCommon<CTaskClass>::m_bForceExit) //forced exit
76 bActive = false; //trigger exit from loop
77
78 else if(CCommon<CTaskClass>::m_qRequest.Delete(pTask) && pTask){ //next task
79 pTask->SetThreadId(m_nThreadId); //set task's thread identifier
80 pTask->Perform(); //perform the task
81 CCommon<CTaskClass>::m_qResult.Insert(pTask); //performed task result
82 } //if
83
84 else bActive = false; //request queue empty, so trigger exit from loop
85 } //while
